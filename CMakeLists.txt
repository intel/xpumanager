cmake_minimum_required (VERSION 3.2.0)

SET(PROJECT_VERSION_MAJOR 1)
SET(PROJECT_VERSION_MINOR 0)

if (NOT DEFINED BUILD_TIMESTAMP)	
	execute_process(
		COMMAND date "+%Y%m%d"
		OUTPUT_VARIABLE BUILD_DATE
		OUTPUT_STRIP_TRAILING_WHITESPACE
	)
	message(STATUS "build date=${BUILD_DATE}")
	SET(BUILD_TIMESTAMP "${BUILD_DATE}")
endif()

if (NOT DEFINED GIT_COMMIT)	
	execute_process(
		COMMAND git rev-parse HEAD
		OUTPUT_VARIABLE BUILD_HASH
		OUTPUT_STRIP_TRAILING_WHITESPACE
	)	
	string(SUBSTRING ${BUILD_HASH} 0 8 GIT_COMMIT)
	message(STATUS "git hash=${GIT_COMMIT}")
endif()


if (NOT DEFINED WORK_PATH)	
	set(WORK_PATH ${CMAKE_CURRENT_BINARY_DIR})	
endif()

set(PROJECT_VERSION "${PROJECT_VERSION_MAJOR}.${PROJECT_VERSION_MINOR}.${BUILD_TIMESTAMP}")
set(XPUM_VERSION_STRING ${PROJECT_VERSION})

project (XPUManager VERSION ${PROJECT_VERSION})

set(CPACK_SET_DESTDIR OFF)
set(CPACK_RPM_PACKAGE_RELOCATABLE ON)

set(CPACK_INLCUDE_INSTALL_DIR "/usr/local/xpum")
set(CPACK_LIB_INSTALL_DIR "/usr/local/lib")
set(CPACK_CONF_INSTALL_DIR "/etc/xpum")
set(CPACK_PACKAGING_INSTALL_PREFIX "/opt/intel/xpum")
set(SYSTEMD_CONFIG_DIR "/lib/systemd/system")
set(SYSTEMD_UNIT_NAME "${CMAKE_CURRENT_LIST_DIR}/daemon/xpum.service")

set(CMAKE_PROJECT_VERSION "${PROJECT_VERSION}")
add_definitions(-DLOADER_VERSION_MAJOR=${PROJECT_VERSION_MAJOR})
add_definitions(-DLOADER_VERSION_MINOR=${PROJECT_VERSION_MINOR})
add_definitions(-DLOADER_VERSION_PATCH=${BUILD_TIMESTAMP})

# Set compilation options
set(CMAKE_CXX_FLAGS "$ENV{CXXFLAGS} -std=c++11 -Wall -pthread")
set(CMAKE_CXX_FLAGS_DEBUG "$ENV{CXXFLAGS} -O0 -g -ggdb")
set(CMAKE_CXX_FLAGS_RELEASE "$ENV{CXXFLAGS} -O3")
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

configure_file("${SYSTEMD_UNIT_NAME}.in" "${SYSTEMD_UNIT_NAME}")

add_subdirectory(core)
add_subdirectory(py)
add_subdirectory(cli)
add_subdirectory(daemon)

if(NOT DEFINED CPACK_GENERATOR)
	if(EXISTS "/etc/debian_version")
		set(CPACK_GENERATOR "DEB")
	elseif(EXISTS "/etc/redhat-release")
		set(CPACK_GENERATOR "RPM")
	elseif(EXISTS "/etc/SUSE-brand" OR EXISTS "/etc/SUSE-release")
		set(CPACK_GENERATOR "RPM")
	else()
		set(CPACK_GENERATOR "ZIP")
	endif()
endif()

message("CPACK_PACKAGE_NAME=${PROJECT_NAME}")
set(CPACK_PACKAGE_CONTACT "Intel Corporation")
set(CPACK_PACKAGE_NAME "${PROJECT_NAME}")
set(CPACK_PACKAGE_FILE_NAME "${PROJECT_NAME}.${PROJECT_VERSION}.${GIT_COMMIT}")
SET(CPACK_PACKAGE_DESCRIPTION "Inter XPU Manager")
set(CPACK_PACKAGE_VERSION_MAJOR "${PROJECT_VERSION_MAJOR}")
set(CPACK_PACKAGE_VERSION_MINOR "${PROJECT_VERSION_MINOR}")
set(CPACK_PACKAGE_VERSION_PATCH "${BUILD_TIMESTAMP}")

if(CPACK_GENERATOR MATCHES "RPM")
	file(WRITE "${CMAKE_BINARY_DIR}/package_file_name" "${CPACK_PACKAGE_FILE_NAME}.rpm")
	set(CPACK_RPM_SPEC_MORE_DEFINE "%global __os_install_post %(echo '%{__os_install_post}' | sed -e 's!/usr/lib[^[:space:]]*/brp-python-bytecompile[[:space:]].*$!!g')")
	set(CPACK_RPM_PACKAGE_NAME "${PROJECT_NAME}.${PROJECT_VERSION}.${GIT_COMMIT}")
	set(CPACK_RPM_FILE_NAME "")
	set(CPACK_RPM_COMPRESSION_TYPE "xz")
	set(CPACK_RPM_PACKAGE_ARCHITECTURE "x86_64")
	set(CPACK_RPM_PACKAGE_AUTOREQ OFF)
	set(CPACK_RPM_PACKAGE_DESCRIPTION "XPU Manager")
	set(CPACK_RPM_PACKAGE_GROUP "System Environment/Libraries")
	set(CPACK_RPM_PACKAGE_LICENSE "MIT")
	set(CPACK_RPM_PACKAGE_RELEASE 1)
	set(CPACK_RPM_PACKAGE_RELEASE_DIST ON)
	set(CPACK_RPM_COMPONENT_INSTALL ON)
	set(CPACK_RPM_PACKAGE_NAME "${PROJECT_NAME}")
	set(CPACK_RPM_EXCLUDE_FROM_AUTO_FILELIST_ADDITION
	  /etc/ld.so.conf.d
	  /usr/local
	  /usr/local/lib64
	  /usr/local/bin
	  /usr/local/include
		/lib
		/usr/lib
	)

	set(CPACK_RPM_POST_INSTALL_SCRIPT_FILE "${CMAKE_CURRENT_LIST_DIR}/postinst")
	set(CPACK_RPM_PRE_UNINSTALL_SCRIPT_FILE "${CMAKE_CURRENT_LIST_DIR}/prerm")
endif()

if(CPACK_GENERATOR MATCHES "DEB")
	file(WRITE "${CMAKE_BINARY_DIR}/package_file_name" "${CPACK_PACKAGE_FILE_NAME}.deb")
	set(CPACK_DEBIAN_PACKAGE_ARCHITECTURE "amd64")
	set(CPACK_DEBIAN_PACKAGE_SHLIBDEPS ON)
	set(CPACK_DEBIAN_XPUM_PACKAGE_NAME "${PROJECT_NAME}")
  set(CPACK_DEB_COMPONENT_INSTALL ON)
  set(CPACK_ARCHIVE_COMPONENT_INSTALL ON)

	#set(CPACK_DEBIAN_PACKAGE_DEPENDS "libc6 (>= 2.3.1-6), libgcc1 (>= 1:3.4.2-12), python2.6, libboost-program-options1.40.0 (>= 1.40.0)")
	set(CPACK_DEBIAN_PACKAGE_CONTROL_EXTRA "${CMAKE_CURRENT_LIST_DIR}/postinst;${CMAKE_SOURCE_DIR}/prerm;")	
endif()

INCLUDE(CPack)

install(
    FILES 
		core/include/xpum_structs.h
		core/include/xpum_api.h
    DESTINATION ${CPACK_INLCUDE_INSTALL_DIR}
)

install(
    FILES 
		core/config/pci.ids
		core/config/pci.switch
    DESTINATION ${CPACK_CONF_INSTALL_DIR}
)


install(TARGETS DGMCore xpumcli xpumd
	LIBRARY DESTINATION ${CPACK_LIB_INSTALL_DIR}
	RUNTIME DESTINATION ${CPACK_PACKAGING_INSTALL_PREFIX}/bin
	ARCHIVE DESTINATION ${CPACK_PACKAGING_INSTALL_PREFIX}
  )

install(
    FILES
		tools/fwUpgrade/GfxFwFPT
    PERMISSIONS
		OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_EXECUTE WORLD_EXECUTE
    DESTINATION ${CPACK_PACKAGING_INSTALL_PREFIX}/bin
)

install(FILES
	"${SYSTEMD_UNIT_NAME}"
	DESTINATION ${SYSTEMD_CONFIG_DIR})
